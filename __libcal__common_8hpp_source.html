64 template <typename T>
65 void pybuffer_check_1D(py::buffer data) {
66  auto log = cal::Logger::get();
67  py::buffer_info info = data.request();
68  std::vector <char> tp = align_format <T> ();
69  bool valid = false;
70  for (auto const & atp : tp) {
71  if (info.format[0] == atp) {
72  valid = true;
73  }
74  }
75  if (!valid) {
76  std::ostringstream o;
77  o << "Python buffer is type '" << info.format
78  << "', which is not in compatible list {";
79  for (auto const & atp : tp) {
80  o << "'" << atp << "',";
81  }
82  o << "}";
83  log.error(o.str().c_str());
84  throw std::runtime_error(o.str().c_str());
85  }
86  if (info.ndim != 1) {
87  std::ostringstream o;
88  o << "Python buffer has " << info.ndim
89  << " dimensions instead of one, shape = ";
90  for (auto const & d : info.shape) {
91  o << d << ", ";
92  }
93  log.error(o.str().c_str());
94  throw std::runtime_error(o.str().c_str());
95  }
96  return;
97 }
