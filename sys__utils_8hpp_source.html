215  public:
216 
217  // type definitions
218  typedef T value_type;
219  typedef T * pointer;

233  };
234 
236  pointer address(reference value) const {
237  return &value;
238  }
239 
240  const_pointer address(const_reference value) const {
241  return &value;
242  }

244  AlignedAllocator() throw() {}
245 
246  AlignedAllocator(AlignedAllocator const &) throw() {}
247 
248  template <typename U>
249  AlignedAllocator(AlignedAllocator <U> const &) throw() {}
250 
251  ~AlignedAllocator() throw() {}
252 
254  size_type max_size() const throw() {
255  return std::numeric_limits <std::size_t>::max() / sizeof(T);
256  }
257 
259  pointer allocate(size_type const num, const void * hint = 0) {
260  pointer align_ptr =
261  static_cast <pointer> (aligned_alloc(num * sizeof(T),
262  SIMD_ALIGN));
263 
264  return align_ptr;
265  }
266 
268  void construct(pointer p, T const & value) {
270  new (static_cast <void *> (p)) T(value);
271  }
272

276  p->~T();
277  }
278

281  aligned_free(static_cast <void *> (p));
