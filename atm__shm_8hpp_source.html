84  T * allocate(int n) {
85 
86 
87  nlocal_ = n / ntasks_;
88 
89  if (nlocal_ * ntasks_ < n) nlocal_ += 1;
90  if (nlocal_ * (rank_ + 1) > n) {
91  nlocal_ = n - nlocal_ * rank_;
92  }
93  if (nlocal_ < 0) nlocal_ = 0;
94 
96  int ret = MPI_Win_allocate_shared(
97  nlocal_ * sizeof(T), sizeof(T), MPI_INFO_NULL, shmcomm_,
98  &local_, &win_);
99  if (ret != MPI_SUCCESS) {
100  if (rank_ == 0) {
101  auto here = cal_HERE();
102  auto log = cal::Logger::get();
103  std::ostringstream o;
104  o << " Failed to allocate " << n / 1024. / 1024.
105  << " MB of shared memory with " << ntasks_
106  << " tasks.";
107  log.error(o.str().c_str(), here);
108  throw std::runtime_error(o.str().c_str());
109  }
110  }
111  n_ = n;
112 
113  MPI_Aint nn;
114  int disp;
115 
120  ret = MPI_Win_shared_query(win_, 0, &nn, &disp, &global_);
121  if (ret != MPI_SUCCESS) {
122  if (rank_ == 0) {
123  auto here = cal_HERE();
124  auto log = cal::Logger::get();
125  std::string msg("Failed to query shared memory address.");
126  log.error(msg.c_str(), here);
127  throw std::runtime_error(msg.c_str());
128  }
129  }
130 
131  return global_;
132  }

135  if (global_) {

137  if (ret != MPI_SUCCESS) {
138  if (rank_ == 0) {
139  auto here = cal_HERE();

141  std::string msg("Failed to free shared memory.");

165  T * resize(int64_t n) {
166 
167  int64_t n_copy = 0;
168  cal::AlignedVector <T> temp;
169 
170  if (n < n_) {
171  // We are shrinking the memory
172  n_copy = n;
173  } else if (n_ > 0) {
174  // We are expanding the memory AND
175  // we currently have a non-zero memory size
176  n_copy = n_;
177  }
178 
179  if ((n_copy > 0) && (rank_ == 0)) {
180  temp.resize(n_copy);
181  std::copy(global_, global_ + n_copy, temp.begin());
182  }
183 
184  free();
185  allocate(n);
186 
187  if ((n_copy > 0) && (rank_ == 0)) {
188  std::copy(temp.begin(), temp.end(), global_);
189  }
190 
191  return global_;
192  }
