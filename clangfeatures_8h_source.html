47 // With clang-3.6, -Wall warns about unused-local-typedefs.
48 // The "obvious" thing to do is to ignore -Wunused-local-typedefs,
49 // but that doesn't work because earlier versions of clang blow
50 // up on an 'unknown warning group'. So we briefly ignore -Wall...
51 // It's tempting to just give up on static assertions in pre-c++11 code.
52 #if !R123_USE_CXX11_STATIC_ASSERT && !defined(R123_STATIC_ASSERT)
53 #define R123_STATIC_ASSERT(expr, msg) \
54 _Pragma("clang diagnostic push") \
55 _Pragma("clang diagnostic ignored \"-Wall\"") \
56 typedef char static_assertion[(!!(expr))*2-1] \
57 _Pragma("clang diagnostic pop")
58 #endif
